function taoBaiKiemTra() {
  var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();

  // ğŸ”¹ Láº¥y dá»¯ liá»‡u tá»« file gá»‘c
  var fileGoc = SpreadsheetApp.openById(DATA_FILE_ID);

  // ğŸ”¹ Cáº­p nháº­t sá»‘ láº§n táº¡o bÃ i kiá»ƒm tra (Ã” A1)
  var soLanTao = sheetGoc.getRange("A1").getValue() || 0;

  // ğŸ”¹ Cáº­p nháº­t ngÃ y táº¡o bÃ i kiá»ƒm tra (Ã” B1)
  var ngayHienTai = new Date();

  // ğŸ”¹ YÃªu cáº§u sá»‘ lÆ°á»£ng tá»«
  var soTu = Browser.inputBox("Nháº­p sá»‘ lÆ°á»£ng tá»« báº¡n muá»‘n láº¥y:", Browser.Buttons.OK_CANCEL);
  soTu = parseInt(soTu, 10);

  var tuVung = data.slice(2); // Loáº¡i bá» 2 dÃ²ng Ä‘áº§u (tiÃªu Ä‘á» vÃ  sá»‘ láº§n táº¡o)

  // ğŸ”¹ Sáº¯p xáº¿p danh sÃ¡ch tá»« theo sá»‘ láº§n há»c (Cá»™t C)
  tuVung.sort((a, b) => (a[2] || 0) - (b[2] || 0));

  var soTuItHoc = Math.ceil(tuVung.length / 2);

  // ğŸ”¹ XÃ³a sheet cÅ© náº¿u cÃ³
